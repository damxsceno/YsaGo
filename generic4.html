LABIRINTO DA MENTE

OBJETIVO: Este jogo envolve perguntas e respostas, e suas escolhas afetam o progresso no labirinto mental. Divirta-se explorando e tomando decisões!

CÓDIGO:

package main
import (
"fmt"
"strings"
)

func main() {
fmt.Println("Bem-vindo ao Labirinto da Mente!")
fmt.Println("Você está prestes a embarcar em uma jornada desafiadora para sair deste labirinto mental.")
fmt.Println("Para avançar, você deve responder a perguntas e tomar decisões. Escolha sabiamente!")

var jogadorNome string
fmt.Print("Antes de começar, qual é o seu nome? ")
fmt.Scanln(&jogadorNome)
fmt.Printf("Ótimo, %s. Vamos começar!\n", jogadorNome)

var resposta string
var avancar bool

// Início do jogo
fmt.Println("\nVocê se encontra em um corredor escuro. À sua frente, há três portas numeradas: 1, 2 e 3.")
fmt.Print("Qual porta você escolhe? ")
fmt.Scanln(&resposta)
if strings.TrimSpace(resposta) == "2" {
fmt.Println("Você escolheu a porta 2 e entrou em uma sala misteriosa.")
fmt.Println("Um enigma é apresentado a você:")
fmt.Println("O que é algo que todos têm, mas ninguém pode perder?")
fmt.Print("Qual é a sua resposta? ")
fmt.Scanln(&resposta)
if strings.ToLower(strings.TrimSpace(resposta)) == "um nome" {
fmt.Println("Você é inteligente, %s! A resposta está correta.", jogadorNome)
avancar = true
} else {
fmt.Println("Resposta incorreta. A porta se fecha. Você continua sua busca.")
}
} else {
fmt.Println("Você escolheu a porta errada e fica preso em um ciclo infinito de corredores.")
}

if avancar {
fmt.Println("\nVocê avança para a próxima sala. Há duas opções à sua frente: 'Escada' ou 'Porta Secreta'.")
fmt.Print("O que você escolhe? ")
fmt.Scanln(&resposta)

if strings.ToLower(strings.TrimSpace(resposta)) == "escada" {
fmt.Println("Você sobe as escadas e encontra um grande livro de quebra-cabeças.")
fmt.Println("Você decide resolver um quebra-cabeça:")
fmt.Println("O que é uma coisa que é sempre quebrada antes de usá-la?")
fmt.Print("Qual é a sua resposta? ")
fmt.Scanln(&resposta)
if strings.ToLower(strings.TrimSpace(resposta)) == "um ovo" {
fmt.Println("Incrível, %s! Você resolveu o quebra-cabeça e continua sua jornada.", jogadorNome)
avancar = true
} else {
fmt.Println("Resposta incorreta. As escadas o levam de volta ao início do labirinto.")
}
} else if strings.ToLower(strings.TrimSpace(resposta)) == "porta secreta" {
fmt.Println("Você abre a porta secreta e entra em uma sala iluminada.")
fmt.Println("Há uma pergunta à sua frente:")
fmt.Println("O que pode correr, mas nunca anda? Tem uma boca, mas nunca fala?")
fmt.Print("Qual é a sua resposta? ")
fmt.Scanln(&resposta)
if strings.ToLower(strings.TrimSpace(resposta)) == "um rio" {
fmt.Printf("Impressionante, %s! Sua resposta está correta e você avança ainda mais no labirinto.", jogadorNome)
avancar = true
} else {
fmt.Println("Resposta incorreta. A sala se fecha e você precisa escolher novamente.")
}
}
}

if avancar {
fmt.Println("\nVocê continua sua jornada e encontra uma bifurcação no caminho.")
fmt.Print("Você escolhe o caminho da esquerda ou da direita? ")
fmt.Scanln(&resposta)
if strings.ToLower(strings.TrimSpace(resposta)) == "esquerda" {
fmt.Println("Você segue o caminho da esquerda e encontra uma chave de ouro brilhante.")
fmt.Println("Com a chave em mãos, você alcança a saída do labirinto.")
fmt.Printf("Parabéns, %s! Você escapou do Labirinto da Mente com sucesso!\n", jogadorNome)
} else if strings.ToLower(strings.TrimSpace(resposta)) == "direita" {
fmt.Println("Você escolhe o caminho da direita e encontra uma sala escura. O labirinto continua...")
} else {
fmt.Println("Você fica paralisado pela indecisão e o labirinto se fecha ao seu redor.")
}
}

fmt.Println("\nObrigado por jogar o Labirinto da Mente! Esperamos que tenha gostado desta jornada desafiadora.")
}

JOGO DA ADIVINHAÇÃO

OBJETIVO: Neste jogo, o computador escolhe um número aleatório entre 0 e 100, e o jogador tenta adivinhar o número. O programa dará dicas se o palpite do jogador for muito alto ou muito baixo. O jogo continuará até que o jogador adivinhe corretamente o número. Divirta-se!

CÓDIGO:

package main
import (
"fmt"
"math/rand"
"time"
)

func main() {
rand.Seed(time.Now().UnixNano())
numeroAleatorio := rand.Intn(100)
tentativas := 0

fmt.Println("Bem-vindo ao Jogo de Adivinhação!")
fmt.Println("Tente adivinhar o número entre 0 e 100.")

for {
var palpite int
fmt.Print("Seu palpite: ")
_, err := fmt.Scanf("%d", &palpite)
if err != nil {
fmt.Println("Por favor, insira um número válido.")
continue
}

tentativas++

if palpite < numeroAleatorio {
fmt.Println("Tente um número maior.")
} else if palpite > numeroAleatorio {
fmt.Println("Tente um número menor.")
} else {
fmt.Printf("Parabéns! Você acertou o número %d em %d tentativas.\n", numeroAleatorio, tentativas)
break
}
}
}

JOGO DE PALAVRAS

OBJETIVO: Neste jogo, o computador escolhe aleatoriamente uma palavra de uma lista e o jogador tenta adivinhar a palavra, fornecendo letras uma de cada vez. O jogador tem um número limitado de tentativas para adivinhar a palavra completa. Divirta-se adivinhando palavras!

CÓDIGO:

package main
import (
"fmt"
"math/rand"
"strings"
"time"
)

func main() {
rand.Seed(time.Now().UnixNano())

palavras := []string{"gato", "cachorro", "pássaro", "elefante", "girafa", "macaco"}
palavraAleatoria := palavras[rand.Intn(len(palavras))]
palavraAdivinhada := strings.Repeat("_", len(palavraAleatoria))
tentativas := 0

fmt.Println("Bem-vindo ao Jogo de Adivinhação de Palavras!")
fmt.Println("Tente adivinhar a palavra oculta.")

for {
fmt.Printf("Palavra atual: %s\n", palavraAdivinhada)
var palpite string
fmt.Print("Digite uma letra: ")
_, err := fmt.Scanf("%s", &palpite)
if err != nil || len(palpite) != 1 {
fmt.Println("Por favor, insira uma única letra válida.")
continue
}

tentativas++
palpite = strings.ToLower(palpite)

novaPalavraAdivinhada := ""
acertou := false

for i, letra := range palavraAleatoria {
if string(letra) == palpite {
novaPalavraAdivinhada += palpite
acertou = true
} else {
novaPalavraAdivinhada += string(palavraAdivinhada[i])
}
}

if !acertou {
fmt.Println("Letra errada. Tente novamente.")
} else {
fmt.Println("Você acertou uma letra!")
}

palavraAdivinhada = novaPalavraAdivinhada

if palavraAdivinhada == palavraAleatoria {
fmt.Printf("Parabéns! Você adivinhou a palavra \"%s\" em %d tentativas.\n", palavraAleatoria, tentativas)
break
}
}
}
